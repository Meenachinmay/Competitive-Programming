#include<iostream>
#include <set>
#include <algorithm>
#include <conio.h>
using namespace std;
int main() {

	set<int> ss;
	int longest = 0;
	int arr[6];
	for (int i = 0; i < 6; i ++) {
		cin >> arr[i];
	}
	for (auto& num : arr) {
		ss.insert(num);
	}
	set<int>::iterator it;
	for (it = ss.begin(); it != ss.end(); it ++) {
		if (!(*it-1)) {
			int curr = *it;
			int count = 1;
			while (ss.count(curr+1)) {
				curr++;
				count++;
			}
			longest = max(longest, count);
		}
	}
	cout << longest << endl;
	_getch();
	return 0;
}
